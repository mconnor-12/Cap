📋 Google Sheets Integration

🔗 Live Connection Settings

This dashboard is prewired to your Google Sheet and deployed Apps Script. On load, it automatically verifies the connection and pulls live student data from the tracker tab.

📜 Full Apps Script — Copy & Deploy

Paste this into Extensions → Apps Script → Deploy → Web App (Anyone). Replace the Sheet ID. This creates new tabs without touching existing ones, enables live point sync, and exposes student data for refresh.
// ═══════════════════════════════════════════════ // AP RESEARCH — MINISTRY OF RESEARCH // Google Apps Script v2 — Full Integration // Deploy as Web App → Anyone can access // ═══════════════════════════════════════════════ const SHEET_ID = '1llsHnCkLPU7mBB2ICt_vb6_aDiW1-Py1tYBpJjsJF1A'; const TRACKER_TAB = 'Data Tracker - Cohort 2026'; const POINTS_TAB = 'House Cup Points'; const MASTERY_TAB = 'Mastery Tracker'; const CONTACTS_TAB = 'Parent Contacts'; const SPRINTS_TAB = 'Writing Sprints'; function doGet(e) { const action = e.parameter.action || 'ping'; const cors = ContentService.createTextOutput; try { let result; if (action === 'ping') result = {ok:true, time: new Date().toISOString()}; else if (action === 'students') result = getStudents(); else if (action === 'points') result = getPoints(); else if (action === 'mastery') result = getMastery(); else if (action === 'contacts') result = getContacts(); else if (action === 'sprints') result = getSprints(); else result = {error: 'Unknown action: ' + action}; return cors(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } catch(err) { return cors(JSON.stringify({error: err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { const cors = ContentService.createTextOutput; try { const data = JSON.parse(e.postData.contents); let result; if (data.action === 'addPoints') result = addPoints(data); else if (data.action === 'addMastery') result = addMastery(data); else if (data.action === 'addContact') result = addContact(data); else if (data.action === 'addSprint') result = addSprint(data); else result = {error: 'Unknown action: ' + data.action}; return cors(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } catch(err) { return cors(JSON.stringify({error: err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function getStudents() { const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ss.getSheetByName(TRACKER_TAB); if (!sh) return {error: 'Tab not found: ' + TRACKER_TAB}; const rows = sh.getDataRange().getValues(); const headers = rows[0].map(h => String(h).trim()); return { students: rows.slice(1) .filter(r => r[1] || r[2]) .map(r => { const obj = {}; headers.forEach((h,i) => obj[h] = r[i]); return obj; }) }; } function getPoints() { const ss = SpreadsheetApp.openById(SHEET_ID); ensureTab(ss, POINTS_TAB, ['Date','House','Student','Achievement','Points','Note','IsBonus']); const sh = ss.getSheetByName(POINTS_TAB); const rows = sh.getDataRange().getValues(); const headers = rows[0]; return { entries: rows.slice(1).map(r => { const o={}; headers.forEach((h,i)=>o[h]=r[i]); return o; }) }; } function getMastery() { const ss = SpreadsheetApp.openById(SHEET_ID); ensureTab(ss, MASTERY_TAB, ['Date','Student','House','Skill','Level','Teacher']); const sh = ss.getSheetByName(MASTERY_TAB); const rows = sh.getDataRange().getValues(); const headers = rows[0]; return { entries: rows.slice(1).map(r => { const o={}; headers.forEach((h,i)=>o[h]=r[i]); return o; }) }; } function getContacts() { const ss = SpreadsheetApp.openById(SHEET_ID); ensureTab(ss, CONTACTS_TAB, ['Date','Student','House','Type','Note','Logged By']); const sh = ss.getSheetByName(CONTACTS_TAB); const rows = sh.getDataRange().getValues(); const headers = rows[0]; return { entries: rows.slice(1).map(r => { const o={}; headers.forEach((h,i)=>o[h]=r[i]); return o; }) }; } function getSprints() { const ss = SpreadsheetApp.openById(SHEET_ID); ensureTab(ss, SPRINTS_TAB, ['Date','Winning House','Points']); const sh = ss.getSheetByName(SPRINTS_TAB); const rows = sh.getDataRange().getValues(); const headers = rows[0]; return { entries: rows.slice(1).map(r => { const o={}; headers.forEach((h,i)=>o[h]=r[i]); return o; }) }; } function addPoints(data) { const ss = SpreadsheetApp.openById(SHEET_ID); ensureTab(ss, POINTS_TAB, ['Date','House','Student','Achievement','Points','Note','IsBonus']); ss.getSheetByName(POINTS_TAB).appendRow([ new Date().toLocaleDateString('en-US'), data.house, data.student||'(House Award)', data.achievement, Number(data.points)||0, data.note||'', data.isBonus?'Yes':'No' ]); return {success: true}; } function addMastery(data) { const ss = SpreadsheetApp.openById(SHEET_ID); ensureTab(ss, MASTERY_TAB, ['Date','Student','House','Skill','Level','Teacher']); ss.getSheetByName(MASTERY_TAB).appendRow([ new Date().toLocaleDateString('en-US'), data.student, data.house, data.skill, data.level, data.teacher||'' ]); return {success: true}; } function addContact(data) { const ss = SpreadsheetApp.openById(SHEET_ID); ensureTab(ss, CONTACTS_TAB, ['Date','Student','House','Type','Note','Logged By']); ss.getSheetByName(CONTACTS_TAB).appendRow([ new Date().toLocaleDateString('en-US'), data.student, data.house, data.type, data.note||'', data.teacher||'' ]); return {success: true}; } function addSprint(data) { const ss = SpreadsheetApp.openById(SHEET_ID); ensureTab(ss, SPRINTS_TAB, ['Date','Winning House','Points']); ss.getSheetByName(SPRINTS_TAB).appendRow([data.date, data.house, 100]); return {success: true}; } function ensureTab(ss, name, headers) { if (!ss.getSheetByName(name)) { const sh = ss.insertSheet(name); sh.appendRow(headers); sh.getRange(1,1,1,headers.length) .setFontWeight('bold') .setBackground('#1a1a2e') .setFontColor('#d4a843'); sh.setFrozenRows(1); } }

📋 Setup Steps

  1. Open your Google Sheet
  2. Click Extensions → Apps Script
  3. Delete existing code, paste the script above
  4. The SHEET_ID is already filled in — no change needed
  5. Click 💾 Save, then Deploy → New Deployment
  6. Type: Web App · Who has access: Anyone · Click Deploy
  7. Copy the Web App URL and paste it in the field above
  8. Click Test — you should see "Connected!"
  9. Five new tabs auto-create: House Cup Points, Mastery Tracker, Parent Contacts, Writing Sprints
  10. Click Pull Data to sync live student data from the tracker tab

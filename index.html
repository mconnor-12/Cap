📜 Full Apps Script — Copy & Deploy

Paste this into Extensions → Apps Script → Deploy → Web App (Anyone). Replace the Sheet ID. This creates new tabs without touching existing ones, enables live point sync, and exposes student data for refresh.
// ═══════════════════════════════════════════════ // AP RESEARCH — MINISTRY OF RESEARCH // Google Apps Script v4 — Claude Artifact Compatible // Deploy as Web App → Anyone can access // Optional: bind to the sheet to use the custom menu // ═══════════════════════════════════════════════ const SHEET_ID = '1llsHnCkLPU7mBB2ICt_vb6_aDiW1-Py1tYBpJjsJF1A'; const TRACKER_TAB = 'Data Tracker - Cohort 2026'; const TRACKER_GID = 1986422992; const POINTS_TAB = 'House Cup Points'; const MASTERY_TAB = 'Mastery Tracker'; const CONTACTS_TAB = 'Parent Contacts'; const SPRINTS_TAB = 'Writing Sprints'; const TAB_HEADERS = { [POINTS_TAB]: ['Date','House','Student','Achievement','Points','Note','IsBonus'], [MASTERY_TAB]: ['Date','Student','House','Skill','Level','Teacher'], [CONTACTS_TAB]: ['Date','Student','House','Type','Note','Logged By'], [SPRINTS_TAB]: ['Date','Winning House','Points'] }; function onOpen() { try { SpreadsheetApp.getUi() .createMenu('Ministry of Research') .addItem('Set Up Required Tabs', 'menuSetupTabs') .addItem('Verify Tabs and Headers', 'menuVerifyWorkbook') .addItem('Repair Headers', 'menuRepairHeaders') .addSeparator() .addItem('Test Students Endpoint', 'menuTestStudents') .addItem('Test Points Endpoint', 'menuTestPoints') .addItem('Test Contacts Endpoint', 'menuTestContacts') .addItem('Test Sprints Endpoint', 'menuTestSprints') .addToUi(); } catch (err) {} } function menuSetupTabs() { const ss = SpreadsheetApp.openById(SHEET_ID); Object.keys(TAB_HEADERS).forEach(name => ensureTab(ss, name, TAB_HEADERS[name])); SpreadsheetApp.getUi().alert('Required tabs are set up.'); } function menuVerifyWorkbook() { SpreadsheetApp.getUi().alert(JSON.stringify(verifyWorkbook(), null, 2)); } function menuRepairHeaders() { const ss = SpreadsheetApp.openById(SHEET_ID); Object.keys(TAB_HEADERS).forEach(name => repairHeaders(ss, name, TAB_HEADERS[name])); SpreadsheetApp.getUi().alert('Managed tab headers repaired.'); } function menuTestStudents() { showJsonDialog('Students', getStudents()); } function menuTestPoints() { showJsonDialog('Points', getPoints()); } function menuTestContacts() { showJsonDialog('Contacts', getContacts()); } function menuTestSprints() { showJsonDialog('Sprints', getSprints()); } function doGet(e) { const action = (e && e.parameter && e.parameter.action) || 'ping'; const callback = e && e.parameter && e.parameter.callback; try { let result; if (action === 'ping') result = { ok: true, time: new Date().toISOString(), spreadsheetId: SHEET_ID }; else if (action === 'students') result = getStudents(); else if (action === 'points') result = getPoints(); else if (action === 'mastery') result = getMastery(); else if (action === 'contacts') result = getContacts(); else if (action === 'sprints') result = getSprints(); else if (action === 'verify') result = verifyWorkbook(); // JSONP-friendly write actions for standalone hosted HTML / Claude artifacts else if (action === 'addPoints') result = addPoints({ house: e.parameter.house || '', student: e.parameter.student || '', achievement: e.parameter.achievement || '', points: Number(e.parameter.points || 0), note: e.parameter.note || '', isBonus: String(e.parameter.isBonus || '').toLowerCase() === 'true' }); else if (action === 'addMastery') result = addMastery({ student: e.parameter.student || '', house: e.parameter.house || '', skill: e.parameter.skill || '', level: e.parameter.level || '', teacher: e.parameter.teacher || '' }); else if (action === 'addContact') result = addContact({ student: e.parameter.student || '', house: e.parameter.house || '', type: e.parameter.type || '', note: e.parameter.note || '', teacher: e.parameter.teacher || e.parameter.loggedBy || '' }); else if (action === 'addSprint') result = addSprint({ date: e.parameter.date || '', house: e.parameter.house || '', points: Number(e.parameter.points || 100) }); else result = { ok: false, error: 'Unknown action: ' + action }; return createOutput(result, callback); } catch (err) { return createOutput({ ok: false, error: String(err), stack: err && err.stack ? err.stack : '' }, callback); } } function doPost(e) { try { const data = JSON.parse((e && e.postData && e.postData.contents) || '{}'); let result; if (data.action === 'addPoints') result = addPoints(data); else if (data.action === 'addMastery') result = addMastery(data); else if (data.action === 'addContact') result = addContact(data); else if (data.action === 'addSprint') result = addSprint(data); else result = { ok: false, error: 'Unknown action: ' + data.action }; return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ ok: false, error: String(err), stack: err && err.stack ? err.stack : '' })) .setMimeType(ContentService.MimeType.JSON); } } function createOutput(obj, callback) { const payload = JSON.stringify(obj); if (callback) { return ContentService .createTextOutput(callback + '(' + payload + ');') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(payload) .setMimeType(ContentService.MimeType.JSON); } function getStudents() { const ss = SpreadsheetApp.openById(SHEET_ID); let sh = ss.getSheetByName(TRACKER_TAB); if (!sh) sh = ss.getSheets().find(s => s.getSheetId() === TRACKER_GID) || null; if (!sh) { return { ok: false, error: 'Tracker tab not found', expectedName: TRACKER_TAB, expectedGid: TRACKER_GID }; } const rows = sh.getDataRange().getValues(); if (!rows.length) return { ok: true, headers: [], students: [], count: 0 }; const headers = rows[0].map(h => String(h || '').trim()); const students = rows.slice(1) .filter(row => row.some(cell => String(cell || '').trim() !== '')) .map(row => { const obj = {}; headers.forEach((h, i) => obj[h] = row[i]); const first = pickField(obj, ['First Name', 'First', 'Student First Name']); const last = pickField(obj, ['Last Name', 'Last', 'Student Last Name']); const full = pickField(obj, ['Student', 'Student Name', 'Full Name', 'Name']) || [first, last].filter(Boolean).join(' ').trim(); const house = pickField(obj, ['House', 'Team', 'Cohort House']); const id = pickField(obj, ['ID', 'Student ID', 'School ID']); obj._studentName = full; obj._house = house; obj._studentId = id; return obj; }); return { ok: true, headers, students, count: students.length }; } function getPoints() { return getEntries(POINTS_TAB, TAB_HEADERS[POINTS_TAB]); } function getMastery() { return getEntries(MASTERY_TAB, TAB_HEADERS[MASTERY_TAB]); } function getContacts() { return getEntries(CONTACTS_TAB, TAB_HEADERS[CONTACTS_TAB]); } function getSprints() { return getEntries(SPRINTS_TAB, TAB_HEADERS[SPRINTS_TAB]); } function getEntries(tabName, headers) { const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ensureTab(ss, tabName, headers); const rows = sh.getDataRange().getValues(); if (!rows.length) return { ok: true, headers, entries: [], count: 0 }; const actualHeaders = rows[0].map(h => String(h || '').trim()); const entries = rows.slice(1) .filter(row => row.some(cell => String(cell || '').trim() !== '')) .map(row => { const obj = {}; actualHeaders.forEach((h, i) => obj[h] = row[i]); return obj; }); return { ok: true, headers: actualHeaders, entries, count: entries.length }; } function addPoints(data) { requireFields(data, ['house', 'achievement']); const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ensureTab(ss, POINTS_TAB, TAB_HEADERS[POINTS_TAB]); sh.appendRow([ new Date().toLocaleDateString('en-US'), data.house || '', data.student || '(House Award)', data.achievement || '', Number(data.points) || 0, data.note || '', truthy(data.isBonus) ? 'Yes' : 'No' ]); return { ok: true, success: true }; } function addMastery(data) { requireFields(data, ['student', 'skill', 'level']); const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ensureTab(ss, MASTERY_TAB, TAB_HEADERS[MASTERY_TAB]); sh.appendRow([ new Date().toLocaleDateString('en-US'), data.student || '', data.house || '', data.skill || '', data.level || '', data.teacher || '' ]); return { ok: true, success: true }; } function addContact(data) { requireFields(data, ['student', 'type']); const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ensureTab(ss, CONTACTS_TAB, TAB_HEADERS[CONTACTS_TAB]); sh.appendRow([ new Date().toLocaleDateString('en-US'), data.student || '', data.house || '', data.type || '', data.note || '', data.teacher || '' ]); return { ok: true, success: true }; } function addSprint(data) { requireFields(data, ['house']); const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ensureTab(ss, SPRINTS_TAB, TAB_HEADERS[SPRINTS_TAB]); sh.appendRow([ data.date || new Date().toLocaleDateString('en-US'), data.house || '', Number(data.points) || 100 ]); return { ok: true, success: true }; } function verifyWorkbook() { const ss = SpreadsheetApp.openById(SHEET_ID); const tracker = ss.getSheetByName(TRACKER_TAB) || ss.getSheets().find(s => s.getSheetId() === TRACKER_GID) || null; const tabs = [{ key: 'tracker', expectedName: TRACKER_TAB, expectedGid: TRACKER_GID, found: !!tracker, actualName: tracker ? tracker.getName() : '', actualGid: tracker ? tracker.getSheetId() : '', headerStatus: 'not enforced' }]; let ok = !!tracker; Object.keys(TAB_HEADERS).forEach(name => { const sh = ss.getSheetByName(name); const found = !!sh; const expectedHeaders = TAB_HEADERS[name]; let actualHeaders = []; let headerMatches = false; if (found && sh.getLastRow() > 0) { actualHeaders = sh.getRange(1, 1, 1, expectedHeaders.length).getValues()[0].map(h => String(h || '').trim()); headerMatches = expectedHeaders.every((h, i) => String(actualHeaders[i]) === String(h)); } if (!found || !headerMatches) ok = false; tabs.push({ key: name, expectedName: name, found, actualName: found ? sh.getName() : '', actualGid: found ? sh.getSheetId() : '', expectedHeaders, actualHeaders, headerMatches }); }); return { ok, spreadsheetId: ss.getId(), spreadsheetName: ss.getName(), checkedAt: new Date().toISOString(), tabs }; } function ensureTab(ss, name, headers) { let sh = ss.getSheetByName(name); if (!sh) sh = ss.insertSheet(name); if (headers && headers.length && sh.getLastRow() === 0) { sh.getRange(1, 1, 1, headers.length).setValues([headers]); styleHeaderRow(sh, headers.length); sh.setFrozenRows(1); } return sh; } function repairHeaders(ss, tabName, headers) { const sh = ensureTab(ss, tabName, headers); sh.getRange(1, 1, 1, headers.length).setValues([headers]); styleHeaderRow(sh, headers.length); sh.setFrozenRows(1); } function styleHeaderRow(sh, colCount) { sh.getRange(1, 1, 1, colCount) .setFontWeight('bold') .setBackground('#1a1a2e') .setFontColor('#d4a843'); } function requireFields(obj, fields) { const missing = fields.filter(f => !obj[f] && obj[f] !== 0 && obj[f] !== false); if (missing.length) throw new Error('Missing required field(s): ' + missing.join(', ')); } function truthy(v) { return v === true || v === 'true' || v === 'TRUE' || v === 'yes' || v === 'Yes' || v === 1 || v === '1'; } function pickField(obj, aliases) { for (let i = 0; i < aliases.length; i++) { const key = aliases[i]; if (obj[key] !== undefined && obj[key] !== null && String(obj[key]).trim() !== '') { return obj[key]; } } return ''; } function showJsonDialog(title, data) { try { const html = HtmlService.createHtmlOutput( '<div style="font-family:Arial,sans-serif;padding:12px;">' + '<h2 style="margin-top:0;">' + escapeHtml(title) + '</h2>' + '<pre style="white-space:pre-wrap;font-size:12px;">' + escapeHtml(JSON.stringify(data, null, 2)) + '</pre></div>' ).setWidth(800).setHeight(600); SpreadsheetApp.getUi().showModalDialog(html, title); } catch (err) {} } function escapeHtml(str) { return String(str) .replace(/&/g, '&amp;') .replace(/</g, '&lt;') .replace(/>/g, '&gt;'); }

📋 Setup Steps

  1. Open your Google Sheet
  2. Click Extensions → Apps Script
  3. Delete existing code, paste the script above
  4. The SHEET_ID is already filled in — no change needed
  5. Click 💾 Save, then Deploy → New Deployment
  6. Type: Web App · Execute as: Me · Who has access: Anyone (NOT "Anyone with a Google Account") · Click Deploy
  7. ⚠️ IMPORTANT for multi-teacher access: "Execute as: Me" + "Anyone" is required so colleagues can read data without being logged into Google. If you previously deployed as "Anyone with a Google Account," redeploy and use the new URL.
  8. Copy the Web App URL and paste it in the field above, then click 💾 Save Override
  9. Click 🔄 Verify Connection — you should see "Connected!"
  10. Five new tabs auto-create: House Cup Points, Mastery Tracker, Parent Contacts, Writing Sprints
  11. Click ⬇️ Pull Live Data to sync live student data from the tracker tab
  12. Share the HTML file URL with your co-teacher — they should click 🔄 Refresh in the nav to pull current data
